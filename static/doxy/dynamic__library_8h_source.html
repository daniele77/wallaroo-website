46 // Manage a shared library as a resource. Load it on the ctor
47 // and unload it on the dtor.
48 // Export a method to get a function pointer given the symbol name.
49 class DynamicLibrary : private PlatformSpecificDynamicLibrary
50 {
51 public:
52  /* Create a DynamicLibrary from the path specified as parameter
53  * loading the OS library.
54  * @param fileName the path of the dynamic library to load
55  * @throw WrongFile if the file does not exist or its format is wrong.
56  */
57  explicit DynamicLibrary( const std::string& fileName ) :
58  PlatformSpecificDynamicLibrary( fileName )
59  {
60  }
61  // Release the OS library
62  ~DynamicLibrary()
63  {
64  }
65  /* Return a function pointer to the symbol @c funcName.
66  * @param funcName the name of the function in the library to load.
67  * @return the function pointer to the symbol required. NULL if the symbol was not found.
68  */
69  template < typename F >
70  F GetFunction( const std::string& funcName )
71  {
72  return PlatformSpecificDynamicLibrary::GetFunction< F >( funcName );
73  }
74  /* Returns the platform-specific filename suffix
75  for shared libraries (including the period).
76  In debug mode, the suffix also includes a
77  "d" to specify the debug version of a library.
78  */
79  static std::string Suffix()
80  {
81  return PlatformSpecificDynamicLibrary::Suffix();
82  }
83 };
