45 // unix-like specific implementation of shared library
46 class PlatformSpecificDynamicLibrary
47 {
48 public:
49  // throw WrongFile if the file does not exist or its format is wrong.
50  explicit PlatformSpecificDynamicLibrary( const std::string& fileName )
51  {
52  libHandle = dlopen( fileName.c_str(), RTLD_LAZY | RTLD_GLOBAL );
53  if ( ! libHandle ) throw WrongFile( fileName );
54  }
55  // Release the OS library
56  ~PlatformSpecificDynamicLibrary()
57  {
58  dlclose( libHandle );
59  }
60  // Return a function pointer to the symbol funcName. nullptr if the symbol was not found.
61  template < typename F >
62  F GetFunction( const std::string& funcName )
63  {
64  F f = (F)dlsym( libHandle, funcName.c_str() );
65  if ( dlerror() != nullptr ) return nullptr;
66  return f;
67  }
68  /* Returns the unix filename suffix
69  for shared libraries (including the period).
70  In debug mode, the suffix also includes a
71  "d" to specify the debug version of a library. */
72  static std::string Suffix()
73  {
74  #if defined(__APPLE__)
75  #if defined(_DEBUG)
76  return "d.dylib";
77  #else
78  return ".dylib";
79  #endif
80  #elif defined(hpux) || defined(_hpux)
81  #if defined(_DEBUG)
82  return "d.sl";
83  #else
84  return ".sl";
85  #endif
86  #elif defined(__CYGWIN__)
87  #if defined(_DEBUG)
88  return "d.dll";
89  #else
90  return ".dll";
91  #endif
92  #else
93  #if defined(_DEBUG)
94  return "d.so";
95  #else
96  return ".so";
97  #endif
98  #endif
99  }
100 private:
101  void* libHandle;
102 };
